@Component
public class CustomWebMvcTagsProvider extends DefaultWebMvcTagsProvider {

    @Override
    public Iterable<Tag> getTags(HttpServletRequest request, HttpServletResponse response, Object handler, Throwable exception) {
        Iterable<Tag> tags = super.getTags(request, response, handler, exception);
        Stream<Tag> tagsStream = StreamSupport.stream(Spliterators.spliteratorUnknownSize(tags.iterator(), Spliterator.ORDERED), false);
        tags = Tags.of(tagsStream.filter(tag -> !("uri".equals(tag.getKey()) && "UNKNOWN".equals(tag.getValue()))).collect(Collectors.toList()));
        String uri = request.getRequestURI();
        return Tags.concat(tags, "uri", uri != null ? uri : "UNKNOWN");
    }

}
