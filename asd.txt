@Component("mdcMessage2Processor")
public class MdcMessage2Processor implements Processor {
    @Override
    public void process(Exchange exchange) {
        String msg = exchange.getIn().getHeader("message2", String.class);
        org.slf4j.MDC.put("message2", msg);
    }
}
